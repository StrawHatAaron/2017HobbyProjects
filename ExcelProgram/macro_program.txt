Sub Button4_Click()    Dim myFile As String, rng As Range, cellValue As Variant, i As Integer, j As Integer    myFile = Application.DefaultFilePath & "\sales.txt"    Set rng = Range("A1:Z5")    Open myFile For Output As #1    For i = 1 To rng.Rows.Count        For j = 1 To rng.Columns.Count            cellValue = rng.Cells(i, j).Value                If j = rng.Columns.Count Then                    Write #1, cellValue                Else                    Write #1, cellValue,                End If        Next j    Next i    Close #1End Sub
